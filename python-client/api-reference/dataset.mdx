<ResponseField name="api" type="LiteralAPI">The API client to handle the dataset.</ResponseField>
<ResponseField name="id" type="str">The unique identifier for the dataset.</ResponseField>
<ResponseField name="created_at" type="str">The timestamp when the dataset was created.</ResponseField>
<ResponseField name="metadata" type="Dict[str, Any]">Additional metadata for the dataset.</ResponseField>
<ResponseField name="name" type="Optional[str]">The name of the dataset. Defaults to None.</ResponseField>
<ResponseField name="description" type="Optional[str]">The description of the dataset. Defaults to None.</ResponseField>
<ResponseField name="items" type="List[DatasetItem]">The items in the dataset. Defaults to [].</ResponseField>
